import State from '../state';
import { ListenAction, highlight } from './utils';

/**
 * highlight function for easier spot the subject
 * @module ListenAction
 */
ListenAction('highlight', (subject_code) => {
  const isHighlight = State.get().highlight[subject_code];
  State.get().highlight.set(subject_code, !isHighlight);
});

ListenAction('resetHighlight', () => {
  State.get().set('highlight', highlight);
});
