import State from '../state';
import Fetch from 'isomorphic-fetch';
import { checkStatus, handleError, ListenAction } from './utils';

/**
 * list all students data for school admin
 * @module ListenAction
*/
ListenAction('list', () => {
  Fetch('/api/list', {
    credentials: 'same-origin'
  })
    .then(checkStatus)
    .then(response => response.json())
    .then(list => {
      State.get().set('list', list);
    })
    // leesei: this won't work since checkStatus() mutated State
    // state.reset('list', list);

    // leesei: better to use reset() for array
    // this won't invalidate current listeners
    // (from doc, no api to listen for specific variable for now)
    .catch(handleError);
});
