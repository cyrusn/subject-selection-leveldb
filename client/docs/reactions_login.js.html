import State from '../state';
import Fetch from 'isomorphic-fetch';
import { checkStatus, handleError, ListenAction, TriggerAction } from './utils';

/**
 * login
 * @module ListenAction
 */
ListenAction('login', (user) => {
  Fetch('/auth/login', {
    credentials: 'same-origin',
    method: 'post',
    headers: {
      'Accept': 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(user)
  })
    .then(checkStatus)
    .then(response => response.json())
    .then((data) => {
      State.get()
        .set('username', user.username)
        .set('data', data)
        .set('isLogin', true);
      return data;
    })
    .then((data) => {
      if (data.isAdmin) {
        TriggerAction('list');
      } else {
        TriggerAction('resetHighlight');
      }
      return data;
    })
    .catch(handleError);
});
