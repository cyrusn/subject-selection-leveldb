import State from '../state';
import Fetch from 'isomorphic-fetch';
import { checkStatus, handleError, ListenAction } from './utils';

/**
 * save choices
 * @module ListenAction
*/
ListenAction('save:choices', () => {
  const choices = State.get().data.choices;

  Fetch('/api/save', {
    credentials: 'same-origin',
    method: 'put',
    headers: {
      'Accept': 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({choices})
  })
    .then(checkStatus)
    .then(response => response.json())
    .catch(handleError);
});
